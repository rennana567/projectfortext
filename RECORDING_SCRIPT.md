```typescript
const analyser = audioContext.createAnalyser();
analyser.fftSize = 2048;

// RMS 音量计算
const calculateRMS = (data: Float32Array) => {
  let sum = 0;
  for (let i = 0; i < data.length; i++) {
    sum += data[i] * data[i];
  }
  return Math.sqrt(sum / data.length);
};
```
